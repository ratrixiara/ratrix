petle i warunki javascript

1. instrukcje warunkowe

przykład jeden: if

let a = 8;
if (a > 5) {
    console.log("a jest większe niż 5");
}

przykład dwa: if / else

let b = 3;
if (b >= 10) {
    console.log("b >= 10");
} else {
    console.log("b < 10");
}

przykład 3: else if

let c = 6;
if (c > 10) {
    console.log("Większe niż 10");
} else if (c > 5) {
    console.log("Większe niż 5");
} else {
    console.log("5 lub mniej");
}

przykład 4: switch

let d = 2;
switch (d) {
    case 1:
        console.log("d jest równe 1");
        break;
    case 2:
        console.log("d jest równe 2");
        break;
    case 3:
        console.log("d jest równe 3");
        break;
    default:
        console.log("d ma inną wartość");
}

2. petle

petla for

for (let i = 0; i < 5; i++) {
    console.log("i =", i);
}

petla do...while

let j = 0;
do {
    console.log("j =", j);
    j++;
} while (j < 3);
